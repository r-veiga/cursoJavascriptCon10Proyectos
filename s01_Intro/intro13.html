Objetos en Javascript

Para crear un objeto puedo asignar de este modo (let o const):

   
            const persona = {
                nombre: 'Zinedine', 
                puesto: 'entrenador', 
                equipos: ['Castilla', 'Real Madrid'], 
                edad: 40, 
                balonDeOro: true
                domicilio {
                    pais: "España", 
                    ciudad: "Madrid"
                }
                saludar: function() { console.log('Hola, soy ' + nombre); }
            }

            console.log(persona.puesto);                    // entrenador
            console.log(persona.equipos[0]);                // Castilla
            console.log(persona.domicilio.ciudad);          // Madrid 
            console.log(persona['domicilio']['ciudad']);    // Madrid
            persona.saludar();                              // Hola, soy Zinedine

Recorrer todas las propiedades de un objeto con un "for"


    const automovil = {
        modelo: 'Camaro', 
        motor: 6.1,
        anio: 1996,
        marca: 'Chevrolet',

        arrancar: function() { console.log('El coche arranca'); }
    };

    for( let propiedad in automovil ) {
        console.log(`${propiedad}: ${automovil[propiedad]}`);
    }

    automovil['arrancar']();  // undefined -- "El coche arranca"
    automovil['modelo']();    // Uncaught TypeError: automovil['modelo'] is not a function
